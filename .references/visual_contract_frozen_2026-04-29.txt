===== R05 — Drawing helpers (f_line_style, f_label_size) =====
f_line_style(string styleName) =>
    styleName == "Dashed" ? line.style_dashed : styleName == "Dotted" ? line.style_dotted : line.style_solid

f_label_size(string sizeName) =>
    sizeName == "Tiny" ? size.tiny : sizeName == "Normal" ? size.normal : sizeName == "Large" ? size.large : size.small

===== R07 — drawAnchoredLine function =====
drawAnchoredLine(line id, bool visible, float price, color col, int width_, int leftBar) =>
    if visible and not na(price)
        int rightBar = extendLevelsRight ? bar_index + tradeLabelOffsetBarsInput : bar_index
        line.set_xy1(id, leftBar, price)
        line.set_xy2(id, rightBar, price)
        line.set_color(id, col)
        line.set_width(id, width_)
        line.set_style(id, fibLineStyle)
        line.set_extend(id, extend.none)
    else
        line.set_color(id, color(na))

===== R08 — setFibLevelLabel function =====
setFibLevelLabel(label idIn, bool visible, float price, string levelName, color col) =>
    label id = idIn
    if visible and isValid and not na(price)
        int labelX = tradeLabelX
        string labelText = levelName + "  " + str.tostring(price, format.mintick)
        if na(id)
            id := label.new(labelX, price, labelText, style=label.style_label_left, color=color.new(color.black, tradeLabelBgTransparencyInput), textcolor=col, size=tradeLabelSize)
        label.set_xy(id, labelX, price)
        label.set_text(id, labelText)
        label.set_color(id, color.new(color.black, tradeLabelBgTransparencyInput))
        label.set_textcolor(id, col)
        label.set_size(id, tradeLabelSize)
    else if not na(id)
        label.delete(id)
        id := na
    id
